189struct mpFloatRect
190{
191 struct
192 {
193 double Xmin;
194 double Xmax;
195 double Ymin;
196 double Ymax;
197 double Y2min;
198 double Y2max;
199 };
200 mpFloatRect() : Xmin(0.0), Xmax(0.0), Ymin(0.0), Ymax(0.0), Y2min(0.0), Y2max(0.0) {}
202 bool PointIsInside(double x, double y) const {
203 if( (x < Xmin || x > Xmax) ||
204 (y < Ymin || y > Ymax) ) return false;
205 return true;
206 }
208 void UpdateBoundingBoxToInclude(double x, double y, double y2=0) {
209 if (x < Xmin ) Xmin = x;
210 else if (x > Xmax ) Xmax = x;
211 if (y < Ymin ) Ymin = y;
212 else if (y > Ymax ) Ymax = y;
213 if (y2 < Y2min) Y2min = y2;
214 else if (y2 > Y2max) Y2max = y2;
215 }
217 void InitializeBoundingBox(double x, double y, double y2=0) {
218 Xmin = Xmax = x;
219 Ymin = Ymax = y;
220 Y2min = Y2max = y2;
221 }
223 bool IsNotSet() const { const mpFloatRect def; return *this==def; }
225 #if (defined(__cplusplus) && (__cplusplus > 201703L)) // C++ > C++17
226 bool operator==(const mpFloatRect&) const = default;
227 #else
228 // We compare with an epsilon precision
229 // NOTE: should be unnecessary as we are looking for any changes; normally this will be an exact match or a real change...
230 bool operator ==(const mpFloatRect &rect) const
231 {
232 auto Same = [this,rect] (double a, double b) { return fabs(a - b) < EPSILON; };
233 return Same(Xmin, rect.Xmin) && Same(Xmax, rect.Xmax) &&
234 Same(Ymin, rect.Ymin) && Same(Ymax, rect.Ymax) &&
235 Same(Y2min, rect.Y2min) && Same(Y2max, rect.Y2max) ;
236 }
237 #endif
